\begin{question}

    In Exercises \ref{proofindfirst} - \ref{proofindlast}, prove each assertion using the Principle of Mathematical Induction.

\begin{problem}\label{proofindfirst}
    $\displaystyle{ \sum_{j=1}^{n} j^2 = \frac{n(n+1)(2n+1)}{6}}$
\end{problem}

\begin{problem}
    $\displaystyle{ \sum_{j=1}^{n} j^3 = \frac{n^2(n+1)^2}{4}}$
\end{problem}

\begin{problem}
    $2^{n} > 500 n$ for $n > 12$
\end{problem}

\begin{problem}
    $3^{n} \geq n^3$ for $n \geq 4$
\end{problem}
 
\begin{problem}
    Use the Product Rule for Absolute Value to show  $\left|x^{n}\right| = |x|^{n}$ for all real numbers $x$ and all natural numbers $n \geq 1$
\end{problem}

\begin{problem}
    Use the Product Rule for Logarithms to show $\log\left(x^{n}\right) = n \log(x)$ for all real numbers $x > 0$ and all natural numbers $n \geq 1$.
\end{problem}

\begin{problem}\label{proofindlast}
    $\left[ \begin{array}{cc} a & 0 \\ 0 & b \\ \end{array} \right]^{n} = \left[ \begin{array}{cc} a^{n} & 0 \\ 0 & b^{n} \\ \end{array} \right]$ for $n \geq 1$.
\end{problem}

\end{question}

\begin{problem}
    
Prove Equations \ref{arithgeoformula} and \ref{arithgeosum} for the case of geometric sequences.  That is:

\begin{enumerate}

\item \label{proofgeosequeneex} For the sequence  $a_{\mbox{\tiny $1$}} = a$, $a_{n\mbox{\tiny{$+1$}}} = r a_{n}$, $n \geq 1$, prove $a_{n} = ar^{n-1}$, $n \geq 1$.

\item  \label{proofgeosumex} $\displaystyle{\sum_{j=1}^{n} a r^{j-1} = a \left( \frac{1-r^n}{1-r}\right)}$, if $r \neq 1$, $\displaystyle{\sum_{j=1}^{n} a r^{j-1} = na}$, if $r=1$.

\end{enumerate}

\end{problem}

\begin{problem}
    
Prove that the determinant of a lower triangular matrix is the product of the entries on the main diagonal.  (See Exercise \ref{triangularmatrices} in Section \ref{MatArithmetic}.)  Use this result to then show $\det\left(I_{n}\right) = 1$ where $I_{n}$ is the $n \times n$ identity matrix.

\end{problem}

\begin{problem}\label{limitpowerruleproof}
    Prove the Power Rule for Limits (see Theorem \ref{LimitProp01} in Section \ref{IntroductiontoLimits}): $\ds{\lim_{x \rightarrow a} \left[f(x)\right]^{n} = \left[\lim_{x \rightarrow a} f(x) \right]^{n}= L^{n}}$, where $n$ is any natural number.
\end{problem}

\begin{problem}
    Discuss the classic  `paradox' \href{http://en.wikipedia.org/wiki/All_horses_are_the_same_color}{\underline{All Horses are the Same Color}} problem with your classmates.
\end{problem}
