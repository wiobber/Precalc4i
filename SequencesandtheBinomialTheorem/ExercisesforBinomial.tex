\begin{question}
    
In Exercises \ref{simpfactfirst} - \ref{simpfactlast},  simplify the given expression.

\begin{problem}\label{simpfactfirst}
    $\left(3!\right)^2$

$\answer{36}$
\end{problem}

\begin{problem}
    $\frac{10!}{7!}$

$\answer{720}$
\end{problem}

\begin{problem}
     $\frac{7!}{2^3 3!}$

    $\answer{105}$
\end{problem}

\begin{problem}
    $\frac{9!}{4! 3! 2!}$

$\answer{1260}$
\end{problem}

\begin{problem}
    $\frac{(n+1)!}{n!}$, $n \geq 0$.

    $\answer{n+1}$
\end{problem}

\begin{problem}
    $\frac{(k-1)!}{(k+2)!}$, $k \geq 1$.

    \begin{solution}
         $\frac{1}{k(k+1)(k+2)}$
    \end{solution}
\end{problem}

\begin{problem}
    $\displaystyle{\binom{8}{3}}$

    $\answer{56}$
\end{problem}

\begin{problem}
     $\displaystyle{\binom{117}{0}}$

     $\answer{1}$
\end{problem}

\begin{problem}\label{simpfactlast}
      $\displaystyle{\binom{n}{n-2}}$, $n \geq 2$

      \begin{solution}
          $\frac{n(n-1)}{2}$
      \end{solution}
\end{problem}

\begin{question}
    
In Exercises \ref{pascalfirst} - \ref{pascallast}, use Pascal's Triangle to expand the given binomial.

\begin{problem} \label{pascalfirst}
    $(x+2)^5$

    \begin{solution}
         $(x+2)^5 = x^5+10x^4+40x^3+80x^2+80x+32$
    \end{solution}
\end{problem}

\begin{problem}
    $(2x-1)^4$
    
    \begin{solution}
        $(2x-1)^4 = 16x^4-32x^3+24x^2-8x+1$
    \end{solution}
\end{problem}

\begin{problem}
    $\left(\frac{1}{3} x +  y^2\right)^3$
    \begin{solution}
    
        $\left(\frac{1}{3} x +  y^2\right)^3 = \frac{1}{27} x^3+\frac{1}{3}x^2y^2+xy^4+y^6$
    \end{solution}
\end{problem}

\begin{problem}\label{pascallast}
    $\left(x - x^{-1} \right)^{4}$
    \begin{solution}
    
        $\left(x - x^{-1} \right)^{4} = x^4-4x^2+6-4x^{-2}+x^{-4}$
    \end{solution}
\end{problem}

\end{question}

\begin{problem}
    You've just won three tickets to see the new film, `$8.\overline{9}$.'  Five of your friends, Albert, Beth, Chuck, Dan, and Eugene, are interested in seeing it with you.  With the help of your classmates, list all the possible ways to distribute your two extra tickets among your five friends.  Now suppose you've come down with the flu.  List all the different ways you can distribute the three tickets among these five friends.  How does this compare with the first list you made?  What does this have to do with the fact that $\binom{5}{2} = \binom{5}{3}$? 
\end{problem}
