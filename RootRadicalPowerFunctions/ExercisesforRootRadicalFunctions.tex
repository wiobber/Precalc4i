\begin {problem}
Use the fact that the $n$th root functions are increasing to solve the following polynomial inequalities:

\begin{itemize}
    \item $x^3 \leq 64$ 

    \begin{solution}
        $(-\infty, 4]$ 
    \end{solution}

    \item $2 - t^5 <  34$

    \begin{solution}
    $(-2,  \infty)$   
    \end{solution}

    \item  $\frac{(2z+1)^3}{4} \geq 2$
    
    \begin{solution}
    $\left[ \frac{1}{2}, \infty \right)$
    \end{solution}

For the following inequalities, remember $\sqrt[n]{x^{n}} = |x|$ if $n$ is even:

    \item  $x^4 \leq 16$
    
    \begin{solution}
    $[-2,2]$
    \end{solution}

    \item $6-t^6 < -58$
    
    \begin{solution}
    $(-\infty, -2) \cup (2, \infty)$
    \end{solution}

    \item $\frac{(2z+1)^4}{3} \geq 27$ 
    
    \begin{solution}
    $(-\infty, -2] \cup [1, \infty)$
    \end{solution}
    
\end{itemize}
\end {problem}

\begin{problem}
According to Einstein's Theory of Special Relativity, the observed mass  of an object is a function of how fast the object is traveling.  Specifically, if  $m_{r}$ is the mass of the object at rest, $v$ is the speed of the object and $c$ is the speed of light, then the observed mass of the object $m(v)$ is given by:
\[m(v) = \dfrac{m_{r}}{\sqrt{1 - \dfrac{v^{2}}{c^{2}}}}\]

\begin{itemize}
\item 
Find the applied domain of the function.

\begin{solution}
$[0, c)$
\end{solution}


\item
Compute $m(.1c), \, m(.5c), \, m(.9c)$ and $m(.999c)$.
\begin{solution}
$m(.1c) = \dfrac{m_{r}}{\sqrt{.99}} \approx 1.005m_{r}$,  $m(.5c) = \dfrac{m_{r}}{\sqrt{.75}} \approx 1.155m_{r}$,  $m(.9c) = \dfrac{m_{r}}{\sqrt{.19}} \approx 2.294m_{r}$, $m(.999c) = \dfrac{m_{r}}{\sqrt{.0.001999}} \approx 22.366m_{r}$.
\end{solution}

\item
Find $ \lim_{v \rightarrow c^{-}} m(v)$.
\begin{solution}
$ \lim_{v \rightarrow c^{-}} m(x) \rightarrow \infty$;   as the object's velocity approaches the speed of light, mass becomes infinite.
\end{solution}

\item
How slowly must the object be traveling so that the observed mass is no greater than 100 times its mass at rest?
\begin{solution}
If the object is traveling no faster than approximately $0.99995$ times the speed of light, then its observed mass will be no greater than $100m_{r}$.
\end{solution}

\end{itemize}
\end {problem}
