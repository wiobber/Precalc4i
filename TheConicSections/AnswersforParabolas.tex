\begin{itemize}

\item  $g(x) =  2 - 2\sqrt{-3x-9}$ represents the lower half of the parabola.

\end{itemize}

\begin{itemize}

\item  $f(x) = -4+2\sqrt{x}$ represents the upper half of the parabola.

\item  $g(x) =  -4 - 2\sqrt{x}$  represents the lower half of the parabola.

\end{itemize}

\begin{itemize}
