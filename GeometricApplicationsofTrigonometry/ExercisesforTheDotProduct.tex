\begin{question}

\begin{itemize}

\end{itemize}

\begin{problem}
A force of $1500$ pounds is required to tow a trailer.  Find the work done towing the trailer along a flat stretch of road $300$ feet.  Assume the force is applied in the direction of the motion.

\begin{solution}
$(1500 \, \text{pounds})(300 \, \text{feet})\cos\left(0^{\circ}\right) =  450,000$ foot-pounds
\end{solution}
\end{problem}

\begin{problem}
Find the work done lifting a $10$ pound book $3$ feet straight up into the air.  Assume the force of gravity is acting straight downwards.

\begin{solution}
$(10 \, \text{pounds})(3 \, \text{feet})\cos\left(0^{\circ}\right) =  30$ foot-pounds
\end{solution}
\end{problem}

\begin{problem}
Suppose Taylor fills her wagon with rocks and must exert a force of 13 pounds to pull her wagon across the yard.  If she maintains a $15^{\circ}$ angle between the handle of the wagon and the horizontal, compute how much work Taylor does pulling her wagon 25 feet.  Round your answer to two decimal places.

\begin{solution}
$(13 \, \text{pounds})(25 \, \text{feet}) \cos\left(15^{\circ}\right) \approx 313.92$ foot-pounds
\end{solution}
\end{problem}

\begin{problem}

\begin{problem}
Find the work done pushing a 200 pound barrel 10 feet up a $12.5^{\circ}$ incline. Ignore all forces acting on the barrel except gravity, which acts downwards.  Round your answer to two decimal places.

\begin{hint}
Since you are working to overcome gravity only, the force being applied acts directly upwards. This means that the angle between the applied force in this case and the motion of the object is \textit{not} the $12.5^{\circ}$ of the incline!
\end{hint}

\begin{solution}
$(200 \, \text{pounds})(10 \, \text{feet}) \cos\left(77.5^{\circ}\right) \approx 432.88$ foot-pounds
\end{solution}
\end{problem}

\begin{problem} \label{triangleineqforvectorsexercise} 
We know that $|x + y| \leq |x| + |y|$ for all real numbers $x$ and $y$ by the Triangle Inequality established in Exercise \ref{triangleinequalityreals} in Section \ref{AbsoluteValueFunctions}.  We can now establish a Triangle Inequality for vectors.  In this exercise, we prove that $\| \overrightarrow{u} + \overrightarrow{v} \| \leq \| \overrightarrow{u} \| + \| \overrightarrow{v} \|$ for all pairs of vectors $\overrightarrow{u}$ and $\overrightarrow{v}$. \index{vector ! triangle inequality}

\begin{enumerate}

\item (Step 1) Show that $\| \overrightarrow{u} + \overrightarrow{v} \|^{2} = \| \overrightarrow{u} \|^{2} + 2\overrightarrow{u} \cdot \overrightarrow{v} + \| \overrightarrow{v} \|^{2}$.

\item (Step 2) Show that $|\overrightarrow{u} \cdot \overrightarrow{v}| \leq \| \overrightarrow{u} \| \| \overrightarrow{v} \|$.  This is the celebrated Cauchy-Schwarz Inequality.\footnote{It is also known by other names.  Check out this \href{http://en.wikipedia.org/wiki/Cauchy-Schwarz_inequality}{\underline{site}} for details.} 

\smallskip

HINT:  Start with $|\overrightarrow{u} \cdot \overrightarrow{v}| = |\; \| \overrightarrow{u} \| \| \overrightarrow{v} \|\cos(\theta) \;|$ and use the fact that $|\cos(\theta)| \leq 1$ for all $\theta$.

\item (Step 3) Show: \[\| \overrightarrow{u} + \overrightarrow{v} \|^{2} = \| \overrightarrow{u} \|^{2} + 2\overrightarrow{u} \cdot \overrightarrow{v} + \| \overrightarrow{v} \|^{2} \leq \| \overrightarrow{u} \|^{2} + 2|\overrightarrow{u} \cdot \overrightarrow{v}| + \| \overrightarrow{v} \|^{2} \leq \| \overrightarrow{u} \|^{2} + 2\| \overrightarrow{u} \| \| \overrightarrow{v} \| + \| \overrightarrow{v} \|^{2} = (\| \overrightarrow{u} \| + \| \overrightarrow{v} \|)^{2}.\]

\item (Step 4) Use Step 3 to show that $\| \overrightarrow{u} + \overrightarrow{v} \| \leq \| \overrightarrow{u} \| + \| \overrightarrow{v} \|$ for all pairs of vectors $\overrightarrow{u}$ and $\overrightarrow{v}$.

\end{enumerate}

\end{problem}
