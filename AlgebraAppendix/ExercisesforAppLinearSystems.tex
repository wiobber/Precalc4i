\begin{question}
In Exercises \ref{reviewsystemfirst} - \ref{reviewsystemlast}, solve the given system using substitution and/or elimination. Classify each system as consistent independent, consistent dependent, or inconsistent. Check your answers both algebraically and graphically.

\begin{problem}\label{reviewsystemfirst} 
$\left\{ \begin{array}{rcr} x+2y & = & 5  \\ x  & = & 6  \end{array} \right.$    
\end{problem}

\begin{problem}
$\left\{ \begin{array}{rcr} 2y-3x & = & 1  \\ y  & = & -3 \end{array} \right.$  
\end{problem}   

\begin{problem}
$\left\{ \begin{array}{rcr} \frac{x+2y}{4} & = & -5  \\ \frac{3x-y}{2}  & = & 1 \end{array} \right.$
\end{problem}

\begin{problem}
$\left\{\begin{array}{ccccc}
      \frac{2}{3} x & - &\frac{1}{5}y&= &3 \\
	 \frac{1}{2}x& +&\frac{3}{4}y&=&1
    \end{array}\right.$
\end{problem}
  
\begin{problem}
$\left\{ \begin{array}{ccccr} \frac{1}{2}x & -& \frac{1}{3}y & = & -1  \\  2y&-&3x & = & 6 \end{array} \right.$   
\end{problem}

\begin{problem}
$\left\{ \begin{array}{ccccr} x&+&4y & = & 6  \\  \frac{1}{12}x&+&\frac{1}{3}y& = & \frac{1}{2}  \end{array} \right.$
\end{problem}
   
\begin{problem}
$\left\{ \begin{array}{ccccr} 3y&-&\frac{3}{2}x & = & -\frac{15}{2}  \\  \frac{1}{2}x&-&y & = & \frac{3}{2} \end{array} \right.$
\end{problem}

\begin{problem}\label{reviewsystemlast}
$\left\{ \begin{array}{ccccr} \frac{5}{6}x&+&\frac{5}{3}y & = & -\frac{7}{3}  \\  -\frac{10}{3}x&-&\frac{20}{3}y & = & 10  \end{array} \right.$
\end{problem} 

\end{question}

\begin{problem}
A local buffet charges $\$7.50$ per person for the basic buffet and $\$9.25$ for the deluxe buffet (which includes crab legs.)  If 27 diners went out to eat and the total bill was $\$227.00$ before taxes, how many chose the basic buffet and how many chose the deluxe buffet?   
\end{problem} 

\begin{problem}
At The Old Home Fill'er Up and Keep on a-Truckin' Cafe, Mavis mixes two different types of coffee beans to produce a house blend.   The first type costs \$3 per pound and the second costs \$8 per pound.  How much of each type does Mavis use to make 50 pounds of a blend which costs \$6 per pound?
\end{problem}

\begin{problem}
Skippy has a total of $\$$10,000 to split between two investments.  One account offers $3\%$ simple interest, and the other account offers $8\%$ simple interest.  For tax reasons, he can only earn $\$500$ in interest the entire year.  How much money should Skippy invest in each account to earn $\$500$ in interest for the year?
\end{problem}

\begin{problem}
A $10 \%$ salt solution is to be mixed with pure water to produce 75 gallons of a $3\%$ salt solution.  How much of each are needed?
\end{problem}

\begin{problem}
This exercise is a follow-up to Example \ref{dudebromixture}.  Work with your classmates to explain why mixing 4 gallons of Sasquatch Sweat\textsuperscript{TM} Energy Drink and 1 gallon of Frooty Giggle Delight\textsuperscript{TM} would also produce a mixture that was ``close enough for the Dude-Bros''.
\end{problem}
