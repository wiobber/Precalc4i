\begin{question}
In Exercises \ref{solveabsvalequfirst} - \ref{solveabsvalequlast}, solve the equation.


\begin{problem}\label{solveabsvalequfirst} 
$|x| = 6$ 

\begin{solution}
    $x = -6$ or $x=6$ 
\end{solution}
\end{problem}

\begin{problem}
$|3t-1| = 10$

\begin{solution}
    $t = -3$ or $t= \dfrac{11}{3}$
\end{solution}
\end{problem}

\begin{problem}
$|4-w| = 7$

\begin{solution}
     $w = -3$ or $w= 11$
\end{solution}
\end{problem}

\begin{problem}
$4 - |y| = 3$

\begin{solution}
    $y = -1$ or $y= 1$
\end{solution}
\end{problem}

\begin{problem}
$2|5m+1| - 3 = 0$

\begin{solution}
    $m=-\dfrac{1}{2}$ or $m= \dfrac{1}{10}$
\end{solution}
\end{problem}

\begin{problem}
$|7x-1| + 2 = 0$

\begin{solution}
    No solution
\end{solution}
\end{problem}

\begin{problem}
$\dfrac{5 - |x|}{2} = 1$

\begin{solution}
    $x=-3$ or $x= 3$
\end{solution}
\end{problem}

\begin{problem}
$\dfrac{2}{3} |5-2w| - \dfrac{1}{2} = 5$

\begin{solution}
    $w = -\dfrac{13}{8}$ or $w= \dfrac{53}{8}$
\end{solution}
\end{problem}

\begin{problem}
$|3t - \sqrt{2}| + 4 = 6$   

\begin{solution}
    $t = \dfrac{\sqrt{2} \pm 2}{3}$
\end{solution}
\end{problem}

\begin{problem}
$\dfrac{|2v+1| - 3}{4} = \dfrac{1}{2} - |2v+1|$  

\begin{solution}
    $v = -1$ or $v = 0$
\end{solution}
\end{problem}

\begin{problem}
$|2x+1| = \dfrac{|2x+1| - 3}{2}$

\begin{solution}
    No solution
\end{solution}
\end{problem}

\begin{problem}
$\dfrac{|3-2y|+ 4}{2} = 2 - |3-2y|$

$y = \answer{\dfrac{3}{2}}$
\end{problem}

\begin{problem}
$|3t - 2| = |2t + 7|$   

\begin{solution}
    $t = -1$ or $t = 9$
\end{solution}
\end{problem}

\begin{problem}
$|3x+1| = |4x|$

\begin{solution}
     $x = -\dfrac{1}{7}$ or $x = 1$
\end{solution}
\end{problem}

\begin{problem}
$|1-\sqrt{2} y| = |y+1|$

\begin{solution}
    $y = 0$ or $y = \dfrac{2}{\sqrt{2} - 1}$
\end{solution}
\end{problem}

\begin{problem}
$|4-x| - |x+2| = 0$

$x = \answer{1}$
\end{problem}

\begin{problem}
$|2-5z| = 5 |z+1|$

$z = \answer{-\dfrac{3}{10}}$
\end{problem}

\begin{problem}\label{solveabsvalequlast}
$\sqrt{3}|w-1| = 2|w+1|$

\begin{solution}
     $w = \dfrac{\sqrt{3} \pm 2}{\sqrt{3} \mp 2}$ 
\end{solution}
\end{problem}

\begin{question}
In Exercises \ref{solveinequabsquadfirst} - \ref{solveinequabsquadlast}, solve the inequality.  Write your answer using interval notation. 

\begin{problem}\label{solveinequabsquadfirst}
$|3x - 5| \leq 4$ 

\begin{solution}
     $\left[\dfrac{1}{3}, 3\right]$
\end{solution}
\end{problem}

\begin{problem}
$|7t + 2| > 10$

\begin{solution}
     $\left(-\infty, -\dfrac{12}{7} \right) \cup \left(\dfrac{8}{7}, \infty\right)$
\end{solution}
\end{problem}

\begin{problem}
$|2w+1| - 5 < 0$

\begin{solution}
    $(-3,2)$
\end{solution}
\end{problem}

\begin{problem}
$|2-y| - 4 \geq -3$

\begin{solution}
    $(-\infty,1] \cup [3,\infty)$
\end{solution}
\end{problem}

\begin{problem}
$|3z+5| + 2 < 1$

\begin{solution}
    No solution
\end{solution}
\end{problem}

\begin{problem}
$2|7-v| +4 > 1$

\begin{solution}
    $(-\infty, \infty)$
\end{solution}
\end{problem}

\begin{problem}
$3 - |x+\sqrt{5}| < -3$

\begin{solution}
     $(-\infty, -6-\sqrt{5}) \cup (6-\sqrt{5}, \infty)$
\end{solution}
\end{problem}

\begin{problem}
$|5t| \leq |t|+3$

\begin{solution}
    $\left[ -\dfrac{3}{4}, \dfrac{3}{4}\right]$
\end{solution}
\end{problem}

\begin{problem}
$|w-3| < |3-w|$

\begin{solution}
    No solution
\end{solution}
\end{problem}

\begin{problem}
$2 \leq |4-y| < 7$

\begin{solution}
     $(-3,2] \cup [6,11)$
\end{solution}
\end{problem}

\begin{problem}
$1 < |2w - 9| \leq 3$

\begin{solution}
     $[3, 4) \cup (5, 6]$
\end{solution}
\end{problem}

\begin{problem}
$3 > 2|\sqrt{3} - x| > 1$ 

\begin{solution}
    $\left(\dfrac{2 \sqrt{3} - 3}{2},  \dfrac{2 \sqrt{3} - 1}{2}   \right) \cup \left(\dfrac{2 \sqrt{3} +1}{2},  \dfrac{2 \sqrt{3} +3}{2}   \right)$
\end{solution}
\end{problem}

\end{question}

\begin{problem}
With help from your classmates, solve:
\begin{enumerate}
\item  $|5 - |2x-3|| = 4$

\begin{solution}
    $x = -3$, or $x = 1$, or $x = 2$, or $x = 6$
\end{solution}

\item   $|5 - |2x-3|| < 4$

\begin{solution}
    $(-3,1) \cup (2,6)$
\end{solution}
 
\end{enumerate}
\end{problem}
