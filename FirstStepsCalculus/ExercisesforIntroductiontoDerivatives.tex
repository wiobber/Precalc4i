\begin{enumerate}

\item  Explain why $g'(0) = \ds{\lim_{\Delta t \rightarrow 0} \frac{g(\Delta t) - g(0)}{\Delta t}}$ does not exist.

\item Find the \index{derivative from the right}\index{derivative ! from the right} derivative \textbf{from the right} at $t=0$:  $g_{+}'(0) = \ds{\lim_{\Delta t \rightarrow 0^{+}} \frac{g(\Delta t) - g(0)}{\Delta t}}$

\item  Find $y = g_{+}'(0) (x-0) + g(0)$ and interpret.

\item Find  $g'(t) =  \ds{\lim_{\Delta t \rightarrow 0} \frac{g(t+\Delta t) - g(t)}{\Delta t}}$.  Assume $t>0$.

\end{enumerate}

\end{problem}

\begin{problem} Let  $f(x) =|x|$.  

\begin{enumerate}

\item  Explain why $f$ is continuous at $x = 0$.

\item\label{cornerex} Show $f'(0)$ does not exist by showing $\ds{\lim_{h \rightarrow 0^{-}} \frac{f(h) - f(0)}{h} =-1}$ but  $\ds{\lim_{h \rightarrow 0^{+}} \frac{f(h) - f(0)}{h} = 1}$.  


        
\item  Graph $y = f(x)$ near $(0,0)$.  Interpret your answer to number \ref{cornerex} graphically.



\item  Find and simplify  $f'(x) =  \ds{\lim_{h \rightarrow 0} \frac{|x+h| - |x|}{h}}$ assuming $x \neq 0$.



\textbf{HINT:}  Consider the two cases $x > 0$ and $x < 0$ \ldots
        


\end{enumerate}

\end{problem}

\begin{problem}  Let  $g(t) = \sqrt[3]{t}$.  

\begin{enumerate}

\item Explain why $g$ is continuous at $t = 0$.

\item\label{verticaltangentex} Show $g'(0)$ does not exist by showing $\ds{\lim_{\Delta t \rightarrow 0} \frac{g(\Delta t) - g(0)}{\Delta t} = \infty}$.  


        
\item  Graph $y = g(t)$ near $(0,0)$.  Interpret your answer to number \ref{verticaltangentex} graphically.



\item  Find and simplify  $g'(t) =  \ds{\lim_{\Delta t \rightarrow 0} \frac{g(t+\Delta t) - g(t)}{\Delta t}}$ assuming $t \neq 0$.



\textbf{HINT:}  $(a-b)\left(a^2+ab+b^2\right) = a^3 - b^3$ 
        


\end{enumerate}

\end{problem}

\begin{problem} Let  $h(x) = x^{\frac{2}{3}}$.  

\begin{enumerate}

\item Explain why $h$ is continuous at $x = 0$.

\item\label{cuspex} Show $h'(0)$ does not exist by showing $\ds{\lim_{\Delta x \rightarrow 0^{-}} \frac{h(\Delta x) - h(0)}{\Delta x} = -\infty}$ and $\ds{\lim_{\Delta x \rightarrow 0^{+}} \frac{h(\Delta x) - h(0)}{\Delta x} = \infty}$


        
\item  Graph $y = h(x)$ near $(0,0)$.  Interpret your answer to number \ref{cuspex} graphically.



\item  Find and simplify  $h'(x) =  \ds{\lim_{\Delta x \rightarrow 0} \frac{h(x+\Delta x) - h(x)}{\Delta x}}$ assuming $x \neq 0$.



\textbf{HINT:}  $(a-b)\left(a^2+ab+b^2\right) = a^3 - b^3$ and $a^2 - b^2 = (a+b)(a-b)$.
        


\end{enumerate}

\end{problem}

\begin{problem}

 Recall from Exercise \ref{JasonHammerExercise1} in Section \ref{QuadraticFunctions} that Carl's friend Jason participates in the Highland Games. In one event, the hammer throw, the height $h(t)$ in feet of the hammer above the ground $t$ seconds after Jason lets it go is modeled by the function $h(t) = -16t^2 +  22.08t + 6$. 

\begin{enumerate}

\item  Find and simplify a formula for the velocity of the hammer, $v(t) = h'(t) = \ds{\lim_{\Delta t \rightarrow 0} \frac{h(t + \Delta t) - h(t)}{\Delta t}}$.

\item  Find and interpret $v(0)$.

\item  Solve $v(t) = 0$ and interpret.

\item  Find the velocity of the hammer when it hits the ground, rounded to three decimal places.

\end{enumerate}

\end{problem}
