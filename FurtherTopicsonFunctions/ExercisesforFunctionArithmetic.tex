\[ f_{+}(x) = \dfrac{f(x) + |f(x)|}{2}, \qquad \text{and} \qquad f_{-}(x) = \dfrac{f(x) - |f(x)|}{2}. \]

\begin{enumerate}

\item Using a graphing utility, graph each of the functions $f$ below along with $f_{+}$ and $f_{-}$.
