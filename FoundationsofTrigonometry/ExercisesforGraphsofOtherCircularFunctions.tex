\begin{question}
In Section \ref{SineCosineLimits}, we observed\footnote{insisted?} that the cosine and sine functions are continuous.  As such, the other four circular functions are continuous on their domains.\footnote{See the remarks following Definition \ref{continuousdefn} in Section \ref{LimitPropertiesandContinuity}.} 
\smallskip
In Exercises \ref{othertriglimitexfirst} - \ref{othertriglimitexlast}, determine the given limit.  Use the symbols `$-\infty$' and `$\infty$'as appropriate.  Check your answers graphically.

\begin{problem}\label{othertriglimitexfirst} 
    $\ds{\lim_{t \rightarrow 0} \tan(t)}$.

    \begin{solution}
        $\ds{\lim_{t \rightarrow 0} \tan(t) = \tan(0) = 0}$.
    \end{solution}
\end{problem}

\begin{problem}
    $\ds{\lim_{t \rightarrow \pi} \sec(t)}$

    \begin{solution}
        $\ds{\lim_{t \rightarrow \pi} \sec(t) = \sec(\pi) = -1}$
    \end{solution}
\end{problem}

\begin{problem}
    $\ds{\lim_{t \rightarrow 3\pi}}$ $\cot\left(\frac{t}{2}\right)$
    
\begin{solution}
    $\ds{\lim_{t \rightarrow 3\pi}}$ $\cot\left(\frac{t}{2}\right) = \cot\left(\frac{3\pi}{2}\right) = 0$
\end{solution}
    
\end{problem}

\begin{problem}
     $\ds{\lim_{\theta \rightarrow 0}}$ $\csc\left(2\theta + \frac{\pi}{4}\right)$.

     \begin{solution}
         $\ds{\lim_{\theta \rightarrow 0}}$ $\csc\left(2\theta + \frac{\pi}{4}\right) = \csc\left(\frac{\pi}{4}\right) = \sqrt{2}$.
     \end{solution}
\end{problem}

\begin{problem}
    $\ds{\lim_{\theta \rightarrow \pi} (\cos(\theta)  -  \sec(\theta))}$

    \begin{solution}
         $\ds{\lim_{\theta \rightarrow \pi} (\cos(\theta)  -  \sec(\theta)) = \cos(\pi) - \sec(\pi) = -1 - (-1) = 0}$
    \end{solution}
\end{problem}

\begin{problem}
    $\ds{\lim_{\theta \rightarrow \frac{\pi}{4}} \sec(\theta) \, \tan(\theta)}$

    \begin{solution}
         $\ds{\lim_{\theta \rightarrow \frac{\pi}{4}} (\sec(\theta) \, \tan(\theta)) = \sec\left( \frac{\pi}{4} \right) \tan\left( \frac{\pi}{4} \right) = \sqrt{2}}$
    \end{solution}
\end{problem}

\begin{problem}
    $\ds{\lim_{x \rightarrow 0^{+}} \csc(3x)}$

    \begin{solution}
        $\ds{\lim_{x \rightarrow 0^{+}} \csc(3x) = \infty}$
    \end{solution}
\end{problem}

\begin{problem}
    $\ds{\lim_{x \rightarrow \pi^{-}}  (\sin(2x)  - \tan(x))}$

    \begin{solution}
        $\ds{\lim_{x \rightarrow \pi^{-}}  (\sin(2x)  - \tan(x))}  = \sin(2\pi) - \tan(\pi) = 0$
    \end{solution}
\end{problem}

\begin{problem}\label{othertriglimitexlast} 
    $\ds{\lim_{x \rightarrow \pi^{+}}}$ $(\cos(x) + \cot(x))$

    \begin{solution}
        $\ds{\lim_{x \rightarrow \pi^{+}}}$ $(\cos(2x) + \cot(x)) =  \infty$
    \end{solution}
\end{problem}

\begin{problem}

In Exercise \ref{sintovertexercise3} in Section \ref{TheOtherCircularFunctions}, we proved $\ds{\lim_{\theta \rightarrow 0}}$ $\frac{\sin(\theta)}{\theta} = 1$.

\begin{enumerate}
\setcounter{enumi}{\value{HW}}

\item \label{thetaoversinetheta} Use Theorem \ref{LimitProp01} from Section \ref{IntroLimits} to show  $\ds{\lim_{\theta \rightarrow 0}}$ $\frac{\theta}{\sin(\theta)} = 1$.

\begin{solution}
    $\ds{\lim_{\theta \rightarrow 0}}$ $\frac{\theta}{\sin(\theta)}$ $ = \ds{\lim_{\theta \rightarrow 0}}$ $\left[\frac{\sin(\theta)}{\theta}\right]^{-1} = 1^{-1} = 1$
\end{solution}
\smallskip


\item  Which indeterminate form is present in the limit $\ds{\lim_{\theta \rightarrow 0^{+}} 2 \theta \, \csc(\theta)}$?

\begin{solution}
    As $\theta \rightarrow 0^{+}$, $2 \theta \, \csc(\theta) \rightarrow 0 \cdot \infty$.
\end{solution}

\smallskip

\item  Graph $f(\theta) = 2 \theta \, \csc(\theta)$ near $\theta = 0$.  What appears to be the limit?

\begin{solution}
    The graph of  $f(\theta) = 2 \theta \, \csc(\theta)$ approaches $(0,2)$ so  $\ds{\lim_{\theta \rightarrow 0^{+}} 2 \theta \, \csc(\theta)}$ appears to be $2$.
\end{solution}

\smallskip

\item  Rewrite $2 \theta \, \csc(\theta)$ in terms of $\theta$ and $\sin(\theta)$ and use part \ref{thetaoversinetheta} to analytically find $\ds{\lim_{\theta \rightarrow 0^{+}} 2 \theta \, \csc(\theta)}$.

\begin{solution}
     $\ds{\lim_{\theta \rightarrow 0^{+}} 2 \theta \, \csc(\theta)}$ $= \ds{\lim_{\theta \rightarrow 0^{+}}}$ $2 \, \theta \frac{1}{\sin(\theta)}$ $= \ds{\lim_{\theta \rightarrow 0^{+}}}$ $2 \, \frac{\theta}{\sin(\theta)} = 2(1) = 2$.
\end{solution}

\item  Use the same methodology as above  to help you analytically determine $\ds{\lim_{\theta \rightarrow 0^{+}} 2 \theta \, \cot(\theta)}$

\begin{solution}
    $\ds{\lim_{\theta \rightarrow 0^{+}} 2 \theta \, \cot(\theta)}$ $= \ds{\lim_{\theta \rightarrow 0^{+}}}$ $2 \, \theta \, \frac{\cos(\theta)}{\sin(\theta)}$  $= \ds{\lim_{\theta \rightarrow 0^{+}}}$ $2 \, \cos(\theta) \, \frac{\theta}{\sin(\theta)} = 2\cos(0)(1) = 2(1)(1) = 2$
\end{solution}


\setcounter{HW}{\value{enumi}}
\end{enumerate}

\end{problem}
