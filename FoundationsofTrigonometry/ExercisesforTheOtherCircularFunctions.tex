\end{solution}
\end{problem}

\end{solution}
\end{problem}

For $T(\theta) =\tan(4 \theta)$, find functions $f$ and $g$ so $T = g \circ f$.

\begin{solution}
One solution is $f(\theta) = 4 \theta$ and $g(\theta) = \tan(\theta)$.
\end{solution}
\end{problem}

 Since $\sec^{2}(\theta) = (\sec(\theta))^2$, one solution is $f(\theta) = \sec(\theta)$ and $g(\theta) = \theta^2$.

\end{solution}
\end{problem}
